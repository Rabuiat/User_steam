CREATE TABLE Usuario(
id_user BIGINT NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
cod_summario BIGINT NOT NULL,
cod_jogo BIGINT NOT NULL,
nome VARCHAR (20) NOT NULL,
senha VARCHAR (30) NOT NULL,
email VARCHAR (50) NOT NULL,
num_trofeus VARCHAR (200) 
);


CREATE TABLE Jogo_user(
id_jogo BIGINT NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
nome_jogo VARCHAR (20) NOT NULL,
temp_jogado DOUBLE 
);


CREATE TABLE Summario(
id_summario BIGINT NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
summario VARCHAR (300)
);


CREATE TABLE Ranking(
id_busca BIGINT NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
cod_user BIGINT NOT NULL,
cod_jogo BIGINT NOT NULL,
porcentagem VARCHAR (50)
);


ALTER TABLE Usuario ADD FOREIGN KEY (cod_summario) REFERENCES Summario (id_summario);
ALTER TABLE Ranking ADD FOREIGN KEY (cod_user) REFERENCES Usuario (id_user);
ALTER TABLE Ranking ADD FOREIGN KEY (cod_jogo) REFERENCES Jogo_user (id_jogo);
ALTER TABLE Usuario ADD FOREIGN KEY (cod_jogo) REFERENCES Jogo_user (id_jogo);
